import { Component } from '@angular/core';
import { NgFor } from '@angular/common';
import { Router, RouterModule } from '@angular/router'; // ← importante!

@Component({
  selector: 'app-ticket-list',
  standalone: true,
  imports: [NgFor, RouterModule], // ← agora sim!
  template: `
    <h2>Tickets</h2> 
    <ul> 
      <li *ngFor="let ticket of tickets">{{ ticket.title }}</li> 
    </ul> 
    <button (click)="irParaUsuarios()">Ir para Usuários</button> 
  `
})
export class TicketListComponent {
  tickets = [
    { title: 'Erro no login' },
    { title: 'Página de perfil não carrega' },
    { title: 'Solicitação de nova funcionalidade' }
  ];

  constructor(private router: Router) {}

  irParaUsuarios() {
     console.log('Botão clicado!');
      this.router.navigate(['/users']);
  }
}
